Arrays and Memory Management

Q1.A.1 — Contiguous Memory

Contiguous memory means data is stored in one single block. Arrays use it so they can find elements quickly. Then we can access any index instantly.

Q1.A.2 — Memory Leaks

A memory leak happens when we take memory from the system but don't give it back. Like when we use new for a dynamic array and forget to use delete[].

Implementation — Linear Search

// Linear Search — O(n) time complexity
int linSrch(int arr[], int size, int key) {
    for (int i = 0; i < size; i++) {
        if (arr[i] == key) {
            return i;
        }
    }
    return -1;
}

#include <iostream>
using namespace std;

// Node structure for a singly linked list
struct Node {
    int data;
    Node* next;
};

class LinkedList {
public:
    Node* head;
    LinkedList() { head = nullptr; }

    void insertAtHead(int val) {
        Node* newNode = new Node();
        newNode->data = val;
        newNode->next = head;
        head = newNode;
    }

    void display() {
        Node* temp = head;
        while (temp != nullptr) {
            cout << temp->data << " -> ";
            temp = temp->next;
        }
        cout << "NULL" << endl;
    }
};

int main() {
    // Demo Execution
    LinkedList list;
    list.insertAtHead(10);
    list.insertAtHead(20);
    list.display();
    return 0;
}
